# 백준 알고리즘 2644
-------------------------------------------
![문제](./img/2644.gng)

~~~ java
import java.util.*; 

public class Main {

    static int[][] familyMap;

    static boolean[] visited;

    static int[] totalCount;

    static int n, x, y;

    public static void main(String args[]) {

      Scanner scan = new Scanner(System.in);

      n = scan.nextInt(); // 전체 사람의 수
   

      familyMap = new int[n+1][2]; // 가계도

      visited = new boolean[n+1];

      totalCount = new int[n+1];
     

      x = scan.nextInt();
      y = scan.nextInt();
    

      int m = scan.nextInt(); // 부모 자식 관계 개수
      int parent, child;
      for(int i=0; i<m; i++){

         parent = scan.nextInt();
         child = scan.nextInt();        

         familyMap[i+1][0] = parent;
         familyMap[i+1][1] = child;

      }    

      System.out.println(BFS()); 

    }   

    public static int BFS(){

        Queue<Integer> q = new LinkedList<>();

        q.add(x);
      
        while(!q.isEmpty()){
           
            int next = q.poll();
            visited[next] = true; // 다녀간 루트 표기
           
            for(int i=1; i<n+1; i++){
                if(next == familyMap[i][0] && !visited[i]){

                    q.add(familyMap[i][1]);
                    totalCount[familyMap[i][1]] = familyMap[i][0]+1;

                }else if(next == familyMap[i][1] && !visited[i]){

                    q.add(familyMap[i][0]);
                    totalCount[familyMap[i][0]] = familyMap[i][1]+1;

                }
            }           

            if(!q.isEmpty() && q.peek() == y){
                return totalCount[y];
            }
        }      

        return -1;
    }
}
~~~
