package bfs_dfs;

import java.util.ArrayList;
import java.util.HashMap;
import java.util.LinkedList;
import java.util.List;
import java.util.Map;
import java.util.Queue;
import java.util.Scanner;

public class num2468 {
	static int n;
	static int maxH = 0;
	static int count = 1;
	static int[][] map;
	static Map<Integer,List<Integer>> listMap = new HashMap<>();
	public static void main(String[] args){
    	Scanner scan = new Scanner(System.in);
    	n = scan.nextInt();
    	map = new int[n][n];
    	
    	int input;
    	for(int i=0; i<n; i++) {
    		for(int j=0; j<n; j++) {
    			input = scan.nextInt();
    			maxH = maxH > input ? maxH : input;    			

    			map[i][j] = input;
    		}
    	}
    	
    	for(int i=1; i<maxH+1; i++) {
    		bfs();
    	}
    	
    	System.out.println(count);
	}
	private static void bfs(List<Integer> list) {		
		
		Queue<Integer> q = new LinkedList<>();
		q.add(list.get(0));
		
		int turn=0;
		int xy, x, y;
		while(!q.isEmpty()) {
			xy = q.poll();
			list.remove(list.contains(xy));
			x = xy/1000;
			y = xy%1000;
			
			turn++;
			
			if(x>0 && list.contains((x-1)*1000+y)) {
				q.add((x-1)*1000+y);
			}
			if(x<n-1 && list.contains((x+1)*1000+y)) {
				q.add((x+1)*1000+y);
			}
			if(y>0 && list.contains(x*1000+(y-1))) {
				q.add(x*1000+(y-1));
			}
			if(y<n-1 && list.contains(x*1000+(y+1))) {
				q.add(x*1000+(y+1));
			}
		}
		count = count < turn ? turn : count;
	}
}
